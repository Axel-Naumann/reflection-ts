\rSec3[temp.dep.type]{Dependent types}

\begin{std.txt}
A type is dependent if it is \newline
[...]

\begin{itemize}
  \item a \grammarterm{simple-template-id} in which either the template name is
  a template parameter or any of the template arguments is a dependent type or
  an expression that is type-dependent or value-dependent, \removed{or}
  \item denoted by \tcode{decltype(}\grammarterm{expression}\tcode{)}, where
        \grammarterm{expression} is type-dependent (14.6.2.2)\removed{.}\added{, or}
  \item \added{denoted by \tcode{reflexpr(}\emph{operand} \tcode{)},
  where \grammarterm{operand} designates a dependent type or a member of an unknown
  specialization.}
\end{itemize}
\end{std.txt}
